sudo nano /etc/systemd/system/videoPlayer.service

aggiungi:

[Unit]

After=network.target

[Service]
ExecStart=/usr/bin/java -jar /home/aricci/Desktop/condivisa/MqttVideoClient/target/mqtt-service-1.0-SNAPSHOT.jar

User=aricci
Restart=always
Environment=DISPLAY=:0
Environment=XAUTHORITY=/home/aricci/.Xauthority

[Install]
WantedBy=multi-user.target


sudo systemctl daemon-reexec
